scene.caption = """In GlowScript programs:

To zoom, drag with middle button or Alt/Option depressed, or use scroll wheel.
On a two-button mouse, middle is left + right.
To pan left/right and up/down, Shift-drag.
Touch screen: pinch/extend to zoom, swipe or two-finger rotate."""


GlowScript 3.0 VPython
scene.forward = vector(0,-.5,-1)

G = 6.7e-11 # Newton gravitational constant

star1 = sphere(pos=vector(-1.9e11,9,0), radius=2e10, color=color.yellow, 
              make_trail=True, trail_type='curve', interval=10)
star1.mass = 2e30
star1.p = vector(0,0,1e4) * star1.mass

star2 = sphere(pos=vector(1.3e11,9,0), radius=2e10, color=color.red,
                make_trail=True, interval=10)
star2.mass = 1e30
star2.p = -star1.p

dt = 1e5
t = 0
while True:
    rate(500)
    # Calculate distance vector between star1 and star2.
    r_vec = star1.pos-star2.pos
    # Calculate magnitude of distance vector.
    r_mag = mag(r_vec)
    # Calcualte unit vector of distance vector.
    r_hat = r_vec/r_mag
    # Calculate force magnitude.
    force_mag = G*star1.mass*star2.mass/r_mag**2
    # Calculate force vector.
    force_vec = -force_mag*r_hat
    
    star1.p = star1.p + force_vec*dt
    star2.p = star2.p - force_vec*dt
    star1.pos = star1.pos + (star1.p/star1.mass) * dt*1.4
    star2.pos = star2.pos + (star2.p/star2.mass) * dt/1.4
    
    t = t+ dt
